# navigate to project folder
cd leetcode

# create a virtual environment
python3 -m venv venv
# this creates a folder: leetcode/venv/

# activate the environment
source venv/bin/activate
# you'll see: (venv) shivam@Shivams-MacBook-Pro leetcode %

# upgrage pip (always do this)
pip install --upgrade pip

# install basic packages
pip install pyspark
pip install pandas

# freeze dependencies
pip freeze > requirements.txt

# add .gitignore (very important)
touch .gitignore
# add this inside
venv/
__pycache__/
*.pyc

# daily workflow (I think VS Code does this automatically)
cd leetcode
source venv/bin/activate
# when done
deactivate


# GIT commands
1. Start from updated main
git checkout main
git pull origin main

2. Create a feature branch
git checkout -b feature/pyspark-practice-1

3. Work and commit there
git add .
git commit -m "describe change"

4. Push feature branch
git push origin feature/pyspark-practice-1

5. Open PR / merge into main
gh pr create ... or merge locally
gh pr create --base main --head feature/pyspark-practice-1 \
--title "PySpark Practice Set 2" \
--body "Added solutions for 6 more PySpark problems"

6. If you want to update your feature from main
git checkout feature/pyspark-practice-1
git pull origin main
(resolve conflicts)
git push origin feature/pyspark-practice-1
